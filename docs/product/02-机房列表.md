# 机房列表页面产品文档

**页面**: `datacenter-list.html`  
**版本**: v2.0  
**日期**: 2026-04-15

---

## 页面概述

机房列表是系统的核心数据管理页面，支持多维度筛选、排序、分页查看机房数据，并提供详情弹窗和导出功能。

---

## 功能模块

### 1. 页面头部

**左侧**: 页面标题 "机房列表"  
**右侧操作按钮**:
- 导出结果: 导出当前筛选结果
- 新增机房: 跳转 datacenter-form.html

#### 2.3 筛选逻辑

```javascript
function applyFilters() {
    // 获取所有筛选条件
    var keyword = document.getElementById('search-keyword').value.toLowerCase().trim();
    var country = document.getElementById('filter-country').value;
    var status = document.getElementById('filter-status').value;
    var kyc = document.getElementById('filter-kyc').value;
    // ... 其他条件
    
    // 过滤数据
    var filtered = currentList.filter(function(d) {
        // 关键词搜索(多字段模糊匹配)
        if (keyword) {
            var fields = [
                (d.name||'').toLowerCase(),
                (d.dc_name||'').toLowerCase(),
                (d.address||'').toLowerCase(),
                (d.park||'').toLowerCase(),
                (d.liaison||'').toLowerCase(),
                (d.source||'').toLowerCase(),
                (d.province||'').toLowerCase(),
                (d.city||'').toLowerCase(),
                (d.contact||'').toLowerCase(),
                (d.remark||'').toLowerCase()
            ];
            var matched = fields.some(function(f){ return f.includes(keyword); });
            if (!matched) return false;
        }
        
        // 精确匹配
        if (country && d.country !== country) return false;
        if (status && d.status !== status) return false;
        if (kyc && d.kycStatus !== kyc) return false;
        
        // MW规模范围匹配
        if (filterMw) {
            var mw = parseFloat(d.availPowerMw) || 0;
            if (filterMw==='micro' && mw >= 1) return false;
            if (filterMw==='small' && (mw < 1 || mw > 5)) return false;
            if (filterMw==='medium' && (mw < 5 || mw > 20)) return false;
            if (filterMw==='large' && (mw < 20 || mw > 100)) return false;
            if (filterMw==='hyperscale' && mw <= 100) return false;
        }
        
        // PUE值范围匹配
        if (pue) {
            var pv = parseFloat(d.pue_value || d.pue) || 0;
            if (pue==='excellent' && pv >= 1.3) return false;
            if (pue==='good' && (pv < 1.3 || pv > 1.4)) return false;
            if (pue==='average' && (pv < 1.4 || pv > 1.5)) return false;
            if (pue==='poor' && pv <= 1.5) return false;
        }
        
        // 交付时间匹配
        if (delivery) {
            var ed = d.expectedDate || d.delivery_date || '';
            if (!ed) {
                if (delivery !== 'immediate') return false;
            } else {
                var now = new Date();
                var target = new Date(ed.indexOf('-') > 0 ? ed : ed + '-01');
                var diffDays = (target - now) / (1000 * 60 * 60 * 24);
                if (delivery === 'immediate' && diffDays > 0) return false;
                if (delivery === '3m' && (diffDays < 0 || diffDays > 90)) return false;
                if (delivery === '6m' && (diffDays < 0 || diffDays > 180)) return false;
                if (delivery === '1y' && (diffDays < 0 || diffDays > 365)) return false;
            }
        }
        
        return true;
    });
    
    // 更新统计
    document.getElementById('record-count').textContent = filtered.length;
    document.getElementById('filter-count').textContent = filtered.length;
    
    // 渲染表格
    currentPage = 1;
    renderTable(filtered.slice(0, PAGE_SIZE));
    _lastFiltered = filtered;
    renderPagination(filtered.length);
}
```

### 4. 排序功能

#### 排序规则

```javascript
var sortKey = 'createdAt';   // 默认排序字段
var sortDir = 'desc';        // 默认降序

function toggleSort(key) {
    if (sortKey === key) {
        sortDir = sortDir === 'asc' ? 'desc' : 'asc';
    } else {
        sortKey = key;
        sortDir = 'desc';
    }
    // 更新表头图标
    // ↕ 默认  ↑ 升序  ↓ 降序
    applyFilters();
}
```

#### 排序优先级

| 字段类型 | 排序方式 |
|----------|----------|
| mw, pue_value | 数值比较 |
| createdAt, delivery_date | 时间戳比较 |
| status | 自定义顺序: new(1) < collected(2) < negotiating(3) < signed(4) < rejected(5) |
| kyc | 自定义顺序: not_started(1) < not_required(2) < in_progress(3) < approved(4) |
| 其他 | 字符串字典序 |

### 5. 分页功能

#### 分页参数

```javascript
var currentPage = 1;
var PAGE_SIZE = 20;  // 每页20条
```

#### 分页计算

```javascript
function renderPagination(total) {
    var totalPages = Math.ceil(total / PAGE_SIZE);
    if (totalPages <= 1) return;
    
    var html = '';
    for (var i = 1; i <= totalPages; i++) {
        html += '<button class="page-btn' + (i === currentPage ? ' active' : '') + '" ' +
                'onclick="goPage(' + i + ')">' + i + '</button>';
    }
    document.getElementById('pagination').innerHTML = html;
}

function goPage(p) {
    currentPage = p;
    var start = (p - 1) * PAGE_SIZE;
    var end = start + PAGE_SIZE;
    
    document.getElementById('page-start').textContent = _lastFiltered.length > 0 ? start + 1 : 0;
    document.getElementById('page-end').textContent = Math.min(end, _lastFiltered.length);
    
    renderTable(_lastFiltered.slice(start, end));
}
```
